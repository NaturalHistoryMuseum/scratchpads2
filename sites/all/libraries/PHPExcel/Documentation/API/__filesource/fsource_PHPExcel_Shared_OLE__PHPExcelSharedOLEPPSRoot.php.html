  136.                     $iSBcnt += floor($raList[$i]->Size / $this->_SMALL_BLOCK_SIZE)

  138.                 } else {
  139.                     $iBBcnt += (floor($raList[$i]->Size / $this->_BIG_BLOCK_SIZE) +
  140.                         (($raList[$i]->Size % $this->_BIG_BLOCK_SIZE)? 1: 0));
  141.                 }
  142.             }
  143.         }
  144.         $iSmallLen = $iSBcnt * $this->_SMALL_BLOCK_SIZE;

  146.         $iSBDcnt = floor($iSBcnt / $iSlCnt) + (($iSBcnt % $iSlCnt)? 1:0);
  147.         $iBBcnt +=  (floor($iSmallLen / $this->_BIG_BLOCK_SIZE) +
  148.                       (( $iSmallLen % $this->_BIG_BLOCK_SIZE)? 1: 0));
  149.         $iCnt = count($raList);

  191.  
  192.         // Calculate BD count

  198.                 $iBdCnt = floor(($iAllW + $iBdCntW) / $iBlCnt) + ((($iAllW+$iBdCntW) % $iBlCnt)? 1: 0);
  199.                 if ($iBdCnt <= ($iBdExL*$iBlCnt+ $i1stBdL)) {
  200.                     break;
  201.                 }
  202.             }
  203.         }
  204.  
  205.         // Save Header
  206.         fwrite($FILE,
  207.                   "\xD0\xCF\x11\xE0\xA1\xB1\x1A\xE1"
  208.                   . "\x00\x00\x00\x00"
  209.                   . "\x00\x00\x00\x00"
  210.                   . "\x00\x00\x00\x00"
  211.                   . "\x00\x00\x00\x00"
  212.                   . pack("v", 0x3b)
  213.                   . pack("v", 0x03)
  214.                   . pack("v", -2)
  215.                   . pack("v", 9)

  217.                   . pack("v", 0)
  218.                   . "\x00\x00\x00\x00"
  219.                   . "\x00\x00\x00\x00"
  220.                   . pack("V", $iBdCnt)
  221.                   . pack("V", $iBBcnt+$iSBDcnt) //ROOT START
  222.                   . pack("V", 0)

  231.                       pack("V", 0)        // Extra BDList Count
  232.                   );
  233.         } else {
  234.             fwrite($FILE, pack("V", $iAll+$iBdCnt) . pack("V", $iBdExL));
  235.         }
  236.  
  237.         // BDList
  238.         for ($i = 0; $i < $i1stBdL && $i < $iBdCnt; ++$i) {
  239.             fwrite($FILE, pack("V", $iAll+$i));
  240.         }
  241.         if ($i < $i1stBdL) {
  242.             $jB = $i1stBdL - $i;
  243.             for ($j = 0; $j < $jB; ++$j) {
  244.                 fwrite($FILE, (pack("V", -1)));
  245.             }
  246.         }
  247.     }
