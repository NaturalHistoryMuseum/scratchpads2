  418.     /**
  419.      * Assemble worksheets into a workbook and send the BIFF data to an OLE
  420.      * storage.
  421.      *
  422.      * @param array $worksheetSizes The sizes in bytes of the binary worksheet streams
  423.      * @return string Binary data for workbook stream
  424.      */
  425.     public function writeWorkbook($pWorksheetSizes = null)
  426.     {
  427.         $this->_worksheetSizes = $pWorksheetSizes;
  428.  
  429.         // Calculate the number of selected worksheet tabs and call the finalization
  430.         // methods for each worksheet
  431.         $total_worksheets = $this->_phpExcel->getSheetCount();
  432.  
  433.         // Add part 1 of the Workbook globals, what goes before the SHEET records
  434.         $this->_storeBof(0x0005);
  435.         $this->_writeCodepage();
  436.         if ($this->_BIFF_version == 0x0600) {
  437.             $this->_writeWindow1();
  438.         }
  439.         if ($this->_BIFF_version == 0x0500) {
  440.             $this->_writeExterns();    // For print area and repeat rows
  441.             $this->_writeNames();      // For print area and repeat rows
  442.         }
  443.         if ($this->_BIFF_version == 0x0500) {
  444.             $this->_writeWindow1();
  445.         }
  446.         $this->_writeDatemode();
  447.         $this->_writeAllFonts();
  448.         $this->_writeAllNumFormats();
  449.         $this->_writeAllXfs();
  450.         $this->_writeAllStyles();
  451.         $this->_writePalette();
  452.  
  453.         // Prepare part 3 of the workbook global stream, what goes after the SHEET records
  454.         $part3 = '';
  455.         if ($this->_country_code != -1) {
  456.             $part3 .= $this->_writeCountry();
  457.         }
  458.         $part3 .= $this->_writeRecalcId();
  459.  
  460.         if ($this->_BIFF_version == 0x0600) {
  461.             $part3 .= $this->_writeSupbookInternal();
  462.             /* TODO: store external SUPBOOK records and XCT and CRN records
  463.             in case of external references for BIFF8 */
  464.             $part3 .= $this->_writeExternsheetBiff8();
  465.             $part3 .= $this->_writeAllDefinedNamesBiff8();
  466.             $part3 .= $this->_writeMsoDrawingGroup();
  467.             $part3 .= $this->_writeSharedStringsTable();
  468.         }
  469.  
  470.         $part3 .= $this->writeEof();
  471.  
  472.         // Add part 2 of the Workbook globals, the SHEET records
  473.         $this->_calcSheetOffsets();
  474.         for ($i = 0; $i < $total_worksheets; ++$i) {
  475.             $this->_writeBoundsheet($this->_phpExcel->getSheet($i), $this->_worksheetOffsets[$i]);
  476.         }
  477.  
  478.         // Add part 3 of the Workbook globals
  479.         $this->_data .= $part3;
  480.  
  481.         return $this->_data;
  482.     }

  488.      */
  489.     function _calcSheetOffsets()
  490.     {
  491.         if ($this->_BIFF_version == 0x0600) {
  492.             $boundsheet_length = 10;  // fixed length for a BOUNDSHEET record
  493.         } else {
  494.             $boundsheet_length = 11;
  495.         }
  496.  
  497.         // size of Workbook globals part 1 + 3
  498.         $offset            = $this->_datasize;
  499.  
  500.         // add size of Workbook globals part 2, the length of the SHEET records
  501.         $total_worksheets = count($this->_phpExcel->getAllSheets());
  502.         foreach ($this->_phpExcel->getWorksheetIterator() as $sheet) {
  503.             if ($this->_BIFF_version == 0x0600) {
  504.                 $offset += $boundsheet_length + strlen(PHPExcel_Shared_String::UTF8toBIFF8UnicodeShort($sheet->getTitle()));

  506.                 $offset += $boundsheet_length + strlen($sheet->getTitle());
  507.             }
  508.         }
  509.  
  510.         // add the sizes of each of the Sheet substreams, respectively
  511.         for ($i = 0; $i < $total_worksheets; ++$i) {
  512.             $this->_worksheetOffsets[$i] = $offset;
  513.             $offset += $this->_worksheetSizes[$i];
  514.         }
  515.         $this->_biffsize = $offset;
  516.     }

  666.     /**
  667.      * Writes all the DEFINEDNAME records (BIFF8).
  668.      * So far this is only used for repeating rows/columns (print titles) and print areas
  669.      */
  670.     private function _writeAllDefinedNamesBiff8()
  671.     {
  672.         $chunk = '';
  673.  
  674.         // Named ranges
  675.         if (count($this->_phpExcel->getNamedRanges()) > 0) {
  676.             // Loop named ranges
  677.             $namedRanges = $this->_phpExcel->getNamedRanges();
  678.             foreach ($namedRanges as $namedRange) {
  679.  
  680.                 // Create absolute coordinate
  681.                 $range = PHPExcel_Cell::splitRange($namedRange->getRange());
  682.                 for ($i = 0; $i < count($range); $i++) {
  683.                     $range[$i][0] = '\'' . str_replace("'", "''", $namedRange->getWorksheet()->getTitle()) . '\'!' . PHPExcel_Cell::absoluteCoordinate($range[$i][0]);
  684.                     if (isset($range[$i][1])) {
  685.                         $range[$i][1] = PHPExcel_Cell::absoluteCoordinate($range[$i][1]);
  686.                     }
  687.                 }
  688.                 $range = PHPExcel_Cell::buildRange($range); // e.g. Sheet1!$A$1:$B$2
  689.  
  690.                 // parse formula
  691.                 try {
  692.                     $error = $this->_parser->parse($range);
  693.                     $formulaData = $this->_parser->toReversePolish();
  694.  
  695.                     // make sure tRef3d is of type tRef3dR (0x3A)
  696.                     if (isset($formulaData{0}) and ($formulaData{0} == "\x7A" or $formulaData{0} == "\x5A")) {
  697.                         $formulaData = "\x3A" . substr($formulaData, 1);
  698.                     }
  699.  
  700.                     if ($namedRange->getLocalOnly()) {
  701.                         // local scope
  702.                         $scope = $this->_phpExcel->getIndex($namedRange->getScope()) + 1;
  703.                     } else {
  704.                         // global scope
  705.                         $scope = 0;
  706.                     }
  707.                     $chunk .= $this->writeData($this->_writeDefinedNameBiff8($namedRange->getName(), $formulaData, $scope, false));
  708.  
  709.                 } catch(Exception $e) {
  710.                     // do nothing
  711.                 }
  712.             }
  713.         }
  714.  
  715.         // total number of sheets
  716.         $total_worksheets = $this->_phpExcel->getSheetCount();
  717.  
  718.         // write the print titles (repeating rows, columns), if any
  719.         for ($i = 0; $i < $total_worksheets; ++$i) {
  720.             $sheetSetup = $this->_phpExcel->getSheet($i)->getPageSetup();
  721.             // simultaneous repeatColumns repeatRows
  722.             if ($sheetSetup->isColumnsToRepeatAtLeftSet() && $sheetSetup->isRowsToRepeatAtTopSet()) {
  723.                 $repeat = $sheetSetup->getColumnsToRepeatAtLeft();
  724.                 $colmin = PHPExcel_Cell::columnIndexFromString($repeat[0]) - 1;
  725.                 $colmax = PHPExcel_Cell::columnIndexFromString($repeat[1]) - 1;
  726.  
  727.                 $repeat = $sheetSetup->getRowsToRepeatAtTop();
  728.                 $rowmin = $repeat[0] - 1;
  729.                 $rowmax = $repeat[1] - 1;
  730.  
  731.                 // construct formula data manually
  732.                 $formulaData = pack('Cv', 0x29, 0x17); // tMemFunc
  733.                 $formulaData .= pack('Cvvvvv', 0x3B, $i, 0, 65535, $colmin, $colmax); // tArea3d
  734.                 $formulaData .= pack('Cvvvvv', 0x3B, $i, $rowmin, $rowmax, 0, 255); // tArea3d
  735.                 $formulaData .= pack('C', 0x10); // tList
  736.  
  737.                 // store the DEFINEDNAME record
  738.                 $chunk .= $this->writeData($this->_writeDefinedNameBiff8(pack('C', 0x07), $formulaData, $i + 1, true));
  739.  
  740.             // (exclusive) either repeatColumns or repeatRows
  741.             } else if ($sheetSetup->isColumnsToRepeatAtLeftSet() || $sheetSetup->isRowsToRepeatAtTopSet()) {
  742.  
  743.                 // Columns to repeat
  744.                 if ($sheetSetup->isColumnsToRepeatAtLeftSet()) {
  745.                     $repeat = $sheetSetup->getColumnsToRepeatAtLeft();
  746.                     $colmin = PHPExcel_Cell::columnIndexFromString($repeat[0]) - 1;
  747.                     $colmax = PHPExcel_Cell::columnIndexFromString($repeat[1]) - 1;
  748.                 } else {
  749.                     $colmin = 0;
  750.                     $colmax = 255;
  751.                 }
  752.                 // Rows to repeat
  753.                 if ($sheetSetup->isRowsToRepeatAtTopSet()) {
  754.                     $repeat = $sheetSetup->getRowsToRepeatAtTop();
  755.                     $rowmin = $repeat[0] - 1;
  756.                     $rowmax = $repeat[1] - 1;
  757.                 } else {
  758.                     $rowmin = 0;
  759.                     $rowmax = 65535;
  760.                 }
  761.  
  762.                 // construct formula data manually because parser does not recognize absolute 3d cell references
  763.                 $formulaData = pack('Cvvvvv', 0x3B, $i, $rowmin, $rowmax, $colmin, $colmax);
  764.  
  765.                 // store the DEFINEDNAME record
  766.                 $chunk .= $this->writeData($this->_writeDefinedNameBiff8(pack('C', 0x07), $formulaData, $i + 1, true));
  767.             }
  768.         }
  769.  
  770.         // write the print areas, if any
  771.         for ($i = 0; $i < $total_worksheets; ++$i) {
  772.             $sheetSetup = $this->_phpExcel->getSheet($i)->getPageSetup();
  773.             if ($sheetSetup->isPrintAreaSet()) {
  774.                 // Print area, e.g. A3:J6,H1:X20
  775.                 $printArea = PHPExcel_Cell::splitRange($sheetSetup->getPrintArea());
  776.                 $countPrintArea = count($printArea);
  777.  
  778.                 $formulaData = '';
  779.                 for ($j = 0; $j < $countPrintArea; ++$j) {
  780.                     $printAreaRect = $printArea[$j]; // e.g. A3:J6
  781.                     $printAreaRect[0] = PHPExcel_Cell::coordinateFromString($printAreaRect[0]);
  782.                     $printAreaRect[1] = PHPExcel_Cell::coordinateFromString($printAreaRect[1]);
  783.  
  784.                     $print_rowmin = $printAreaRect[0][1] - 1;
  785.                     $print_rowmax = $printAreaRect[1][1] - 1;
  786.                     $print_colmin = PHPExcel_Cell::columnIndexFromString($printAreaRect[0][0]) - 1;
