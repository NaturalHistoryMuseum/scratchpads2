  65.      * Implements support for fopen().
  66.      * For creating streams using this wrapper, use OLE_PPS_File::getStream().
  67.      * @param  string  resource name including scheme, e.g.
  68.      *                  ole-chainedblockstream://oleInstanceId=1
  69.      * @param  string  only "r" is supported
  70.      * @param  int     mask of STREAM_REPORT_ERRORS and STREAM_USE_PATH
  71.      * @param  string  absolute path of the opened stream (out parameter)
  72.      * @return bool    true on success
  73.      */
  74.     public function stream_open($path, $mode, $options, &$openedPath)
  75.     {
  76.         if ($mode != 'r') {
  77.             if ($options & STREAM_REPORT_ERRORS) {
  78.                 trigger_error('Only reading is supported', E_USER_WARNING);
  79.             }
  80.             return false;
  81.         }
  82.  
  83.         // 25 is length of "ole-chainedblockstream://"
  84.         parse_str(substr($path, 25), $this->params);
  85.         if (!isset($this->params['oleInstanceId'],
  86.                    $this->params['blockId'],
  87.                    $GLOBALS['_OLE_INSTANCES'][$this->params['oleInstanceId']])) {
  88.  
  89.             if ($options & STREAM_REPORT_ERRORS) {
  90.                 trigger_error('OLE stream not found', E_USER_WARNING);
  91.             }
  92.             return false;
  93.         }
  94.         $this->ole = $GLOBALS['_OLE_INSTANCES'][$this->params['oleInstanceId']];
  95.  
  96.         $blockId = $this->params['blockId'];
  97.         $this->data = '';
  98.         if (isset($this->params['size']) &&
  99.             $this->params['size'] < $this->ole->bigBlockThreshold &&
  100.             $blockId != $this->ole->root->_StartBlock) {
  101.  
  102.             // Block id refers to small blocks
  103.             $rootPos = $this->ole->_getBlockOffset($this->ole->root->_StartBlock);
  104.             while ($blockId != -2) {
  105.                 $pos = $rootPos + $blockId * $this->ole->bigBlockSize;
  106.                 $blockId = $this->ole->sbat[$blockId];
  107.                 fseek($this->ole->_file_handle, $pos);
  108.                 $this->data .= fread($this->ole->_file_handle, $this->ole->bigBlockSize);
  109.             }
  110.         } else {
  111.             // Block id refers to big blocks
  112.             while ($blockId != -2) {
  113.                 $pos = $this->ole->_getBlockOffset($blockId);
  114.                 fseek($this->ole->_file_handle, $pos);
  115.                 $this->data .= fread($this->ole->_file_handle, $this->ole->bigBlockSize);
  116.                 $blockId = $this->ole->bbat[$blockId];
  117.             }
  118.         }
  119.         if (isset($this->params['size'])) {
  120.             $this->data = substr($this->data, 0, $this->params['size']);
  121.         }
  122.  
  123.         if ($options & STREAM_USE_PATH) {
  124.             $openedPath = $path;
  125.         }
  126.  
  127.         return true;
  128.     }

  187.     public function stream_seek($offset, $whence)
  188.     {
  189.         if ($whence == SEEK_SET && $offset >= 0) {

  195.         } else {
  196.             return false;
  197.         }
  198.         return true;
