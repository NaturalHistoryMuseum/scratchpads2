  189.     /**
  190.      * Write SheetViews
  191.      *
  192.      * @param    PHPExcel_Shared_XMLWriter            $objWriter        XML Writer
  193.      * @param    PHPExcel_Worksheet                    $pSheet            Worksheet
  194.      * @throws    Exception
  195.      */
  196.     private function _writeSheetViews(PHPExcel_Shared_XMLWriter $objWriter = null, PHPExcel_Worksheet $pSheet = null)
  197.     {
  198.         // sheetViews
  199.         $objWriter->startElement('sheetViews');
  200.  
  201.             // Sheet selected?
  202.             $sheetSelected = false;
  203.             if ($this->getParentWriter()->getPHPExcel()->getIndex($pSheet) == $this->getParentWriter()->getPHPExcel()->getActiveSheetIndex())
  204.                 $sheetSelected = true;
  205.  
  206.  
  207.             // sheetView
  208.             $objWriter->startElement('sheetView');
  209.             $objWriter->writeAttribute('tabSelected',        $sheetSelected ? '1' : '0');
  210.             $objWriter->writeAttribute('workbookViewId',    '0');
  211.  
  212.                 // Zoom scales
  213.                 if ($pSheet->getSheetView()->getZoomScale() != 100) {
  214.                     $objWriter->writeAttribute('zoomScale',    $pSheet->getSheetView()->getZoomScale());
  215.                 }
  216.                 if ($pSheet->getSheetView()->getZoomScaleNormal() != 100) {
  217.                     $objWriter->writeAttribute('zoomScaleNormal',    $pSheet->getSheetView()->getZoomScaleNormal());
  218.                 }
  219.  
  220.                 // Gridlines
  221.                 if ($pSheet->getShowGridlines()) {
  222.                     $objWriter->writeAttribute('showGridLines',    'true');
  223.                 } else {
  224.                     $objWriter->writeAttribute('showGridLines',    'false');
  225.                 }
  226.  
  227.                 // Row and column headers
  228.                 if ($pSheet->getShowRowColHeaders()) {
  229.                     $objWriter->writeAttribute('showRowColHeaders', '1');
  230.                 } else {
  231.                     $objWriter->writeAttribute('showRowColHeaders', '0');
  232.                 }
  233.  
  234.                 // Right-to-left
  235.                 if ($pSheet->getRightToLeft()) {
  236.                     $objWriter->writeAttribute('rightToLeft',    'true');
  237.                 }
  238.  
  239.                 $activeCell = $pSheet->getActiveCell();
  240.  
  241.                 // Pane
  242.                 $pane = '';
  243.                 $topLeftCell = $pSheet->getFreezePane();
  244.                 if (($topLeftCell != '') && ($topLeftCell != 'A1')) {
  245.                     $activeCell = $topLeftCell;
  246.                     // Calculate freeze coordinates
  247.                     $xSplit = $ySplit = 0;
  248.  
  249.                     list($xSplit, $ySplit) = PHPExcel_Cell::coordinateFromString($topLeftCell);
  250.                     $xSplit = PHPExcel_Cell::columnIndexFromString($xSplit);
  251.  
  252.                     // pane
  253.                     $pane = 'topRight';
  254.                     $objWriter->startElement('pane');
  255.                     if ($xSplit > 1)
  256.                         $objWriter->writeAttribute('xSplit',    $xSplit - 1);
  257.                     if ($ySplit > 1) {
  258.                         $objWriter->writeAttribute('ySplit',    $ySplit - 1);
  259.                         $pane = ($xSplit > 1) ? 'bottomRight' : 'bottomLeft';
  260.                     }
  261.                     $objWriter->writeAttribute('topLeftCell',    $topLeftCell);
  262.                     $objWriter->writeAttribute('activePane',    $pane);
  263.                     $objWriter->writeAttribute('state',        'frozen');
  264.                     $objWriter->endElement();
  265.  
  266.                     if (($xSplit > 1) && ($ySplit > 1)) {
  267.                         //    Write additional selections if more than two panes (ie both an X and a Y split)
  268.                         $objWriter->startElement('selection');    $objWriter->writeAttribute('pane', 'topRight');        $objWriter->endElement();
  269.                         $objWriter->startElement('selection');    $objWriter->writeAttribute('pane', 'bottomLeft');    $objWriter->endElement();
  270.                     }
  271.                 }
  272.  
  273.                 // Selection
  274.                 if ($pane != '') {
  275.                     //    Only need to write selection element if we have a split pane
  276.                     //        We cheat a little by over-riding the active cell selection, setting it to the split cell
  277.                     $objWriter->startElement('selection');
  278.                     $objWriter->writeAttribute('pane', $pane);
  279.                     $objWriter->writeAttribute('activeCell', $activeCell);
  280.                     $objWriter->writeAttribute('sqref', $activeCell);
  281.                     $objWriter->endElement();
  282.                 }
  283.  
  284.             $objWriter->endElement();
  285.  
  286.         $objWriter->endElement();
  287.     }

  289.     /**
  290.      * Write SheetFormatPr
  291.      *
  292.      * @param    PHPExcel_Shared_XMLWriter $objWriter        XML Writer
  293.      * @param    PHPExcel_Worksheet          $pSheet            Worksheet
  294.      * @throws    Exception
  295.      */
  296.     private function _writeSheetFormatPr(PHPExcel_Shared_XMLWriter $objWriter = null, PHPExcel_Worksheet $pSheet = null)
  297.     {
  298.         // sheetFormatPr
  299.         $objWriter->startElement('sheetFormatPr');
  300.  
  301.             // Default row height
  302.             if ($pSheet->getDefaultRowDimension()->getRowHeight() >= 0) {
  303.                 $objWriter->writeAttribute('customHeight',        'true');
  304.                 $objWriter->writeAttribute('defaultRowHeight',    PHPExcel_Shared_String::FormatNumber($pSheet->getDefaultRowDimension()->getRowHeight()));
  305.             } else {
  306.                 $objWriter->writeAttribute('defaultRowHeight', '12.75');
  307.             }
  308.  
  309.             // Default column width
  310.             if ($pSheet->getDefaultColumnDimension()->getWidth() >= 0) {
  311.                 $objWriter->writeAttribute('defaultColWidth', PHPExcel_Shared_String::FormatNumber($pSheet->getDefaultColumnDimension()->getWidth()));
  312.             }
  313.  
  314.             // Outline level - row
  315.             $outlineLevelRow = 0;
  316.             foreach ($pSheet->getRowDimensions() as $dimension) {
  317.                 if ($dimension->getOutlineLevel() > $outlineLevelRow) {
  318.                     $outlineLevelRow = $dimension->getOutlineLevel();
  319.                 }
  320.             }
  321.             $objWriter->writeAttribute('outlineLevelRow',    (int)$outlineLevelRow);
  322.  
  323.             // Outline level - column
  324.             $outlineLevelCol = 0;
  325.             foreach ($pSheet->getColumnDimensions() as $dimension) {
  326.                 if ($dimension->getOutlineLevel() > $outlineLevelCol) {
  327.                     $outlineLevelCol = $dimension->getOutlineLevel();
  328.                 }
  329.             }
  330.             $objWriter->writeAttribute('outlineLevelCol',    (int)$outlineLevelCol);
  331.  
  332.         $objWriter->endElement();
  333.     }
  334.  
  335.     /**
  336.      * Write Cols
  337.      *
  338.      * @param    PHPExcel_Shared_XMLWriter            $objWriter        XML Writer
  339.      * @param    PHPExcel_Worksheet                    $pSheet            Worksheet
  340.      * @throws    Exception
  341.      */
  342.     private function _writeCols(PHPExcel_Shared_XMLWriter $objWriter = null, PHPExcel_Worksheet $pSheet = null)
  343.     {
  344.         // cols
  345.         if (count($pSheet->getColumnDimensions()) > 0)  {
  346.             $objWriter->startElement('cols');
  347.  
  348.                 $pSheet->calculateColumnWidths();
  349.  
  350.                 // Loop through column dimensions
  351.                 foreach ($pSheet->getColumnDimensions() as $colDimension) {
  352.                     // col
  353.                     $objWriter->startElement('col');
  354.                     $objWriter->writeAttribute('min',    PHPExcel_Cell::columnIndexFromString($colDimension->getColumnIndex()));
  355.                     $objWriter->writeAttribute('max',    PHPExcel_Cell::columnIndexFromString($colDimension->getColumnIndex()));
  356.  
  357.                     if ($colDimension->getWidth() < 0) {
  358.                         // No width set, apply default of 10
  359.                         $objWriter->writeAttribute('width',        '9.10');
  360.                     } else {
  361.                         // Width set
  362.                         $objWriter->writeAttribute('width',        PHPExcel_Shared_String::FormatNumber($colDimension->getWidth()));
  363.                     }
  364.  
  365.                     // Column visibility
  366.                     if ($colDimension->getVisible() == false) {
  367.                         $objWriter->writeAttribute('hidden',        'true');
  368.                     }
  369.  
  370.                     // Auto size?
  371.                     if ($colDimension->getAutoSize()) {
  372.                         $objWriter->writeAttribute('bestFit',        'true');
  373.                     }
  374.  
  375.                     // Custom width?
  376.                     if ($colDimension->getWidth() != $pSheet->getDefaultColumnDimension()->getWidth()) {
  377.                         $objWriter->writeAttribute('customWidth',    'true');
  378.                     }
  379.  
  380.                     // Collapsed
  381.                     if ($colDimension->getCollapsed() == true) {
  382.                         $objWriter->writeAttribute('collapsed',        'true');
  383.                     }
  384.  
  385.                     // Outline level
  386.                     if ($colDimension->getOutlineLevel() > 0) {
  387.                         $objWriter->writeAttribute('outlineLevel',    $colDimension->getOutlineLevel());
  388.                     }
  389.  
  390.                     // Style
  391.                     $objWriter->writeAttribute('style', $colDimension->getXfIndex());
  392.  
  393.                     $objWriter->endElement();
  394.                 }
  395.  
  396.             $objWriter->endElement();
  397.         }
  398.     }

  400.     /**
  401.      * Write SheetProtection
  402.      *
  403.      * @param    PHPExcel_Shared_XMLWriter            $objWriter        XML Writer
  404.      * @param    PHPExcel_Worksheet                    $pSheet            Worksheet

  407.     private function _writeSheetProtection(PHPExcel_Shared_XMLWriter $objWriter = null, PHPExcel_Worksheet $pSheet = null)
  408.     {
  409.         // sheetProtection
  410.         $objWriter->startElement('sheetProtection');
  411.  
  412.         if ($pSheet->getProtection()->getPassword() != '') {
  413.             $objWriter->writeAttribute('password',                $pSheet->getProtection()->getPassword());
  414.         }
  415.  
  416.         $objWriter->writeAttribute('sheet',                ($pSheet->getProtection()->getSheet()                ? 'true' : 'false'));
  417.         $objWriter->writeAttribute('objects',                ($pSheet->getProtection()->getObjects()            ? 'true' : 'false'));
  418.         $objWriter->writeAttribute('scenarios',            ($pSheet->getProtection()->getScenarios()            ? 'true' : 'false'));
