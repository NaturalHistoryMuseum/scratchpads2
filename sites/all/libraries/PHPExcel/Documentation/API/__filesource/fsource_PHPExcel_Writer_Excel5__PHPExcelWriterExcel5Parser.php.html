  834.             }

  836.             if ($sheet2 == -1) {
  837.                 throw new Exception("Unknown sheet name $sheet_name2 in formula");
  838.             }

  850.         }

  853.         $offset = -1 - $sheet1;
  854.

  858.     /**

  860.      * (or range). If it doesn't exist yet add it to the workbook's references
  861.      * array. It assumes all sheet names given must exist.
  862.      *
  863.      * @access private
  864.      * @param string $ext_ref The name of the external reference
  865.      * @return mixed The reference index in packed() format on success
  866.      */
  867.     function _getRefIndex($ext_ref)
  868.     {
  869.         $ext_ref = preg_replace("/^'/", '', $ext_ref); // Remove leading  ' if any.
  870.         $ext_ref = preg_replace("/'$/", '', $ext_ref); // Remove trailing ' if any.
  871.         $ext_ref = str_replace('\'\'', '\'', $ext_ref); // Replace escaped '' with '
  872.  
  873.         // Check if there is a sheet range eg., Sheet1:Sheet2.
  874.         if (preg_match("/:/", $ext_ref)) {
  875.             list($sheet_name1, $sheet_name2) = explode(':', $ext_ref);
  876.  
  877.             $sheet1 = $this->_getSheetIndex($sheet_name1);
  878.             if ($sheet1 == -1) {
  879.                 throw new Exception("Unknown sheet name $sheet_name1 in formula");
  880.             }
  881.             $sheet2 = $this->_getSheetIndex($sheet_name2);
  882.             if ($sheet2 == -1) {
  883.                 throw new Exception("Unknown sheet name $sheet_name2 in formula");
  884.             }
  885.  
  886.             // Reverse max and min sheet numbers if necessary
  887.             if ($sheet1 > $sheet2) {
  888.                 list($sheet1, $sheet2) = array($sheet2, $sheet1);
  889.             }
  890.         } else { // Single sheet name only.
  891.             $sheet1 = $this->_getSheetIndex($ext_ref);
  892.             if ($sheet1 == -1) {
  893.                 throw new Exception("Unknown sheet name $ext_ref in formula");
  894.             }
  895.             $sheet2 = $sheet1;
  896.         }
  897.  
  898.         // assume all references belong to this document
  899.         $supbook_index = 0x00;
  900.         $ref = pack('vvv', $supbook_index, $sheet1, $sheet2);
  901.         $total_references = count($this->_references);
  902.         $index = -1;

  1030.     /**

  1167.                 }

  1170.                        !preg_match("/[0-9]/",$this->_lookahead) and
  1171.                        ($this->_lookahead != ':') and ($this->_lookahead != '.'))
  1172.                 {
  1173.                     return $token;
  1174.                 }
  1175.                 // if it's a range A1:A2 or $A$1:$A$2

  1217.     }

  1237.     /**

  1256.             $this->_advance();

  1263.         } elseif ($this->_current_token == "=") {
