  232.     /**
  233.      * Calculate an (approximate) OpenXML column width, based on font size and text contained
  234.      *
  235.      * @param     int        $fontSize            Font size (in pixels or points)
  236.      * @param     bool    $fontSizeInPixels    Is the font size specified in pixels (true) or in points (false) ?
  237.      * @param     string    $cellText            Text to calculate width
  238.      * @param     int        $rotation            Rotation angle
  239.      * @return     int        Column width
  240.      */
  241.     public static function calculateColumnWidth(PHPExcel_Style_Font $font, $cellText = '', $rotation = 0, PHPExcel_Style_Font $defaultFont = null) {
  242.  
  243.         // If it is rich text, use plain text
  244.         if ($cellText instanceof PHPExcel_RichText) {
  245.             $cellText = $cellText->getPlainText();
  246.         }
  247.  
  248.         // Special case if there are one or more newline characters ("\n")
  249.         if (strpos($cellText, "\n") !== false) {
  250.             $lineTexts = explode("\n", $cellText);
  251.             $lineWitdhs = array();
  252.             foreach ($lineTexts as $lineText) {
  253.                 $lineWidths[] = self::calculateColumnWidth($font, $lineText, $rotation = 0, $defaultFont);
  254.             }
  255.             return max($lineWidths); // width of longest line in cell
  256.         }
  257.  
  258.         // Try to get the exact text width in pixels
  259.         try {
  260.             // If autosize method is set to 'approx', use approximation
  261.             if (self::$autoSizeMethod == self::AUTOSIZE_METHOD_APPROX) {
  262.                 throw new Exception('AutoSize method is set to approx');
  263.             }
  264.  
  265.             // Width of text in pixels excl. padding
  266.             $columnWidth = self::getTextWidthPixelsExact($cellText, $font, $rotation);
  267.  
  268.             // Excel adds some padding, use 1.07 of the width of an 'n' glyph
  269.             $columnWidth += ceil(self::getTextWidthPixelsExact('0', $font, 0) * 1.07); // pixels incl. padding
  270.  
  271.         } catch (Exception $e) {
  272.             // Width of text in pixels excl. padding, approximation
  273.             $columnWidth = self::getTextWidthPixelsApprox($cellText, $font, $rotation);
  274.  
  275.             // Excel adds some padding, just use approx width of 'n' glyph
  276.             $columnWidth += self::getTextWidthPixelsApprox('n', $font, 0);
  277.         }
  278.  
  279.         // Convert from pixel width to column width
  280.         $columnWidth = PHPExcel_Shared_Drawing::pixelsToCellDimension($columnWidth, $defaultFont);
  281.  
  282.         // Return
  283.         return round($columnWidth, 6);
  284.     }

  286.     /**
  287.      * Get GD text width in pixels for a string of text in a certain font at a certain rotation angle
  288.      *
  289.      * @param string $text 
  290.      * @param PHPExcel_Style_Font 
  291.      * @param int $rotation 
  292.      * @return int 
  293.      * @throws Exception
  294.      */
  295.     public static function getTextWidthPixelsExact($text, PHPExcel_Style_Font $font, $rotation = 0) {
  296.         if (!function_exists('imagettfbbox')) {
  297.             throw new Exception('GD library needs to be enabled');
  298.         }
  299.  
  300.         // font size should really be supplied in pixels in GD2,
  301.         // but since GD2 seems to assume 72dpi, pixels and points are the same
  302.         $fontFile = self::getTrueTypeFontFileFromFont($font);
  303.         $textBox = imagettfbbox($font->getSize(), $rotation, $fontFile, $text);
  304.  
  305.         // Get corners positions
  306.         $lowerLeftCornerX  = $textBox[0];
  307.         $lowerLeftCornerY  = $textBox[1];
  308.         $lowerRightCornerX = $textBox[2];
  309.         $lowerRightCornerY = $textBox[3];
  310.         $upperRightCornerX = $textBox[4];
  311.         $upperRightCornerY = $textBox[5];
  312.         $upperLeftCornerX  = $textBox[6];
  313.         $upperLeftCornerY  = $textBox[7];
  314.  
  315.         // Consider the rotation when calculating the width
  316.         $textWidth = max($lowerRightCornerX - $upperLeftCornerX, $upperRightCornerX - $lowerLeftCornerX);
  317.  
  318.         return $textWidth;
  319.     }

  321.     /**
  322.      * Get approximate width in pixels for a string of text in a certain font at a certain rotation angle
  323.      *
  324.      * @param string $columnText 
  325.      * @param PHPExcel_Style_Font $font 
  326.      * @param int $rotation 
  327.      * @return int Text width in pixels (no padding added)
  328.      */
  329.     public static function getTextWidthPixelsApprox($columnText, PHPExcel_Style_Font $font = null, $rotation = 0)
  330.     {
  331.         $fontName = $font->getName();
  332.         $fontSize = $font->getSize();
  333.  
  334.         // Calculate column width in pixels. We assume fixed glyph width. Result varies with font name and size.
  335.         switch ($fontName) {
  336.             case 'Calibri':
  337.                 // value 8.26 was found via interpolation by inspecting real Excel files with Calibri 11 font.
  338.                 $columnWidth = (int) (8.26 * PHPExcel_Shared_String::CountCharacters($columnText));
  339.                 $columnWidth = $columnWidth * $fontSize / 11; // extrapolate from font size
  340.                 break;
  341.  
  342.             case 'Arial':
  343.                 // value 7 was found via interpolation by inspecting real Excel files with Arial 10 font.
  344.                 $columnWidth = (int) (7 * PHPExcel_Shared_String::CountCharacters($columnText));
  345.                 $columnWidth = $columnWidth * $fontSize / 10; // extrapolate from font size
  346.                 break;
  347.  
  348.             case 'Verdana':
  349.                 // value 8 was found via interpolation by inspecting real Excel files with Verdana 10 font.
  350.                 $columnWidth = (int) (8 * PHPExcel_Shared_String::CountCharacters($columnText));
