  421.             }

  427.                 $this->_writeString($row, $column, $cell->getValue()->getPlainText(), $xfIndex);
  428.             } else {
  429.                 switch ($cell->getDatatype()) {
  430.

  433.                         $this->_writeBlank($row, $column, $xfIndex);
  434.                     } else {
  435.                         $this->_writeString($row, $column, $cell->getValue(), $xfIndex);
  436.                     }
  437.                     break;

  440.                     $calculatedValue = $this->_preCalculateFormulas ?
  441.                         $cell->getCalculatedValue() : null;
  442.                     $this->_writeFormula($row, $column, $cell->getValue(), $xfIndex, $calculatedValue);
  443.                     break;
  444.  
  445.                 case PHPExcel_Cell_DataType::TYPE_BOOL:
  446.                     $this->_writeBoolErr($row, $column, $cell->getValue(), 0, $xfIndex);
  447.                     break;
  448.  
  449.                 case PHPExcel_Cell_DataType::TYPE_ERROR:
  450.                     $this->_writeBoolErr($row, $column, $this->_mapErrorCode($cell->getValue()), 1, $xfIndex);
  451.                     break;
  452.  
  453.                 case PHPExcel_Cell_DataType::TYPE_NUMERIC:
  454.                     $this->_writeNumber($row, $column, $cell->getValue(), $xfIndex);
  455.                     break;
  456.                 }
  457.             }
  458.         }
  459.  
  460.         // Append
  461.         if ($this->_BIFF_version == 0x0600) {
  462.             $this->_writeMsoDrawing();
  463.         }
  464.         $this->_writeWindow2();
  465.         $this->_writeZoom();
  466.         if ($this->_phpSheet->getFreezePane()) {
  467.             $this->_writePanes();
  468.         }
  469.         $this->_writeSelection();
  470.         $this->_writeMergedCells();
  471.  
  472.         // Hyperlinks
  473.         if ($this->_BIFF_version == 0x0600) {
  474.             foreach ($this->_phpSheet->getHyperLinkCollection() as $coordinate => $hyperlink) {
  475.                 list($column, $row) = PHPExcel_Cell::coordinateFromString($coordinate);
  476.  
  477.                 $url = $hyperlink->getUrl();
  478.  
  479.                 if ( strpos($url, 'sheet://') !== false ) {
  480.                     // internal to current workbook
  481.                     $url = str_replace('sheet://', 'internal:', $url);
  482.  
  483.                 } else if ( preg_match('/^(http:|https:|ftp:|mailto:)/', $url) ) {
  484.                     // URL
  485.                     // $url = $url;
  486.  
  487.                 } else {
  488.                     // external (local file)
  489.                     $url = 'external:' . $url;
  490.                 }
  491.  
  492.                 $this->_writeUrl($row - 1, PHPExcel_Cell::columnIndexFromString($column) - 1, $url);
  493.             }
  494.         }
  495.  
  496.         if ($this->_BIFF_version == 0x0600) {
  497.             $this->_writeDataValidity();
  498.             $this->_writeSheetLayout();
  499.             $this->_writeSheetProtection();
  500.             $this->_writeRangeProtection();
  501.         }
  502.  
  503.         $this->_storeEof();
  504.     }

  506.     /**
  507.      * Write a cell range address in BIFF8
  508.      * always fixed range
  509.      * See section 2.5.14 in OpenOffice.org's Documentation of the Microsoft Excel File Format
  510.      *
  511.      * @param string $range E.g. 'A1' or 'A1:B6'
  512.      * @return string Binary data
  513.      */
  514.     private function _writeBIFF8CellRangeAddressFixed($range = 'A1')
  515.     {
  516.         $explodes = explode(':', $range);
  517.  
  518.         // extract first cell, e.g. 'A1'
  519.         $firstCell = $explodes[0];
  520.  
  521.         // extract last cell, e.g. 'B6'
  522.         if (count($explodes) == 1) {
  523.             $lastCell = $firstCell;
  524.         } else {
  525.             $lastCell = $explodes[1];
  526.         }
  527.  
  528.         $firstCellCoordinates = PHPExcel_Cell::coordinateFromString($firstCell); // e.g. array(0, 1)
  529.         $lastCellCoordinates  = PHPExcel_Cell::coordinateFromString($lastCell);  // e.g. array(1, 6)
  530.  
  531.         return(pack('vvvv',
  532.             $firstCellCoordinates[1] - 1,
  533.             $lastCellCoordinates[1] - 1,
  534.             PHPExcel_Cell::columnIndexFromString($firstCellCoordinates[0]) - 1,
  535.             PHPExcel_Cell::columnIndexFromString($lastCellCoordinates[0]) - 1
  536.         ));
  537.     }

  541.      * sized chunks.

  543.      * @return string The data
  544.      */

  546.     {
  547.         $buffer = 4096;

  551.             $tmp   = $this->_data;
  552.             unset($this->_data);
  553.             return $tmp;
  554.         }
  555.         // No data to return

  557.     }

  592.     /**
  593.      * Write a double to the specified row and column (zero indexed).
  594.      * An integer can be written as a double. Excel will display an
  595.      * integer. $format is optional.
  596.      *
  597.      * Returns  0 : normal termination
  598.      *         -2 : row or column out of range
  599.      *
  600.      * @param integer $row    Zero indexed row
  601.      * @param integer $col    Zero indexed column
  602.      * @param float   $num    The number to write
  603.      * @param mixed   $format The optional XF format
  604.      * @return integer 
  605.      */
  606.     private function _writeNumber($row, $col, $num, $xfIndex)
  607.     {
  608.         $record    = 0x0203;                 // Record identifier
  609.         $length    = 0x000E;                 // Number of bytes to follow
  610.  
  611.         $header    = pack("vv",  $record, $length);
  612.         $data      = pack("vvv", $row, $col, $xfIndex);
  613.         $xl_double = pack("d",   $num);
  614.         if (PHPExcel_Writer_Excel5_BIFFwriter::getByteOrder()) { // if it's Big Endian
  615.             $xl_double = strrev($xl_double);
  616.         }
  617.  
  618.         $this->_append($header.$data.$xl_double);
  619.         return(0);
  620.     }
  621.  
  622.     /**
  623.      * Write a LABELSST record or a LABEL record. Which one depends on BIFF version
  624.      *

  626.      * @param int $col Column index (0-based)
  627.      * @param string $str The string
  628.      * @param int $xfIndex Index to XF record
  629.      */
  630.     private function _writeString($row, $col, $str, $xfIndex)
  631.     {

  637.     }
  638.     /**
  639.      * Write a string to the specified row and column (zero indexed).
  640.      * NOTE: there is an Excel 5 defined limit of 255 characters.
  641.      * $format is optional.
  642.      * Returns  0 : normal termination
  643.      *         -2 : row or column out of range
  644.      *         -3 : long string truncated to 255 chars
  645.      *
  646.      * @access public
  647.      * @param integer $row    Zero indexed row
  648.      * @param integer $col    Zero indexed column
  649.      * @param string  $str    The string to write
  650.      * @param mixed   $format The XF format for the cell
  651.      * @return integer 
  652.      */
  653.     private function _writeLabel($row, $col, $str, $xfIndex)
  654.     {
  655.         $strlen    = strlen($str);
  656.         $record    = 0x0204;                   // Record identifier
  657.         $length    = 0x0008 + $strlen;         // Bytes to follow
  658.  
  659.         $str_error = 0;
  660.  
  661.         if ($strlen > $this->_xls_strmax) { // LABEL must be < 255 chars
  662.             $str       = substr($str, 0, $this->_xls_strmax);
  663.             $length    = 0x0008 + $this->_xls_strmax;
  664.             $strlen    = $this->_xls_strmax;
  665.             $str_error = -3;
  666.         }
  667.  
  668.         $header    = pack("vv",   $record, $length);
  669.         $data      = pack("vvvv", $row, $col, $xfIndex, $strlen);
  670.         $this->_append($header . $data . $str);
  671.         return($str_error);
  672.     }
  673.  
  674.     /**
  675.      * Write a string to the specified row and column (zero indexed).
  676.      * This is the BIFF8 version (no 255 chars limit).
  677.      * $format is optional.
  678.      * Returns  0 : normal termination
  679.      *         -2 : row or column out of range
  680.      *         -3 : long string truncated to 255 chars
  681.      *
  682.      * @access public
  683.      * @param integer $row    Zero indexed row
  684.      * @param integer $col    Zero indexed column
  685.      * @param string  $str    The string to write
  686.      * @param mixed   $format The XF format for the cell
  687.      * @return integer 
  688.      */
  689.     private function _writeLabelSst($row, $col, $str, $xfIndex)
  690.     {
  691.         $record    = 0x00FD;                   // Record identifier
  692.         $length    = 0x000A;                   // Bytes to follow
  693.  
  694.         $str = PHPExcel_Shared_String::UTF8toBIFF8UnicodeLong($str);
  695.  
  696.         /* check if string is already present */
  697.         if (!isset($this->_str_table[$str])) {
  698.             $this->_str_table[$str] = $this->_str_unique++;
  699.         }
  700.         $this->_str_total++;
  701.  
  702.         $header    = pack('vv',   $record, $length);
  703.         $data      = pack('vvvV', $row, $col, $xfIndex, $this->_str_table[$str]);
  704.         $this->_append($header.$data);
  705.     }
  706.  
  707.     /**
  708.      * Writes a note associated with the cell given by the row and column.
  709.      * NOTE records don't have a length limit.
  710.      *
  711.      * @param integer $row    Zero indexed row
  712.      * @param integer $col    Zero indexed column
  713.      * @param string  $note   The note to write
  714.      */
  715.     private function _writeNote($row, $col, $note)
  716.     {
  717.         $note_length    = strlen($note);
  718.         $record         = 0x001C;                // Record identifier
  719.         $max_length     = 2048;                  // Maximun length for a NOTE record
  720.         //$length      = 0x0006 + $note_length;    // Bytes to follow
  721.  
  722.         // Length for this record is no more than 2048 + 6
  723.         $length    = 0x0006 + min($note_length, 2048);
  724.         $header    = pack("vv",   $record, $length);
  725.         $data      = pack("vvv", $row, $col, $note_length);
  726.         $this->_append($header . $data . substr($note, 0, 2048));
  727.  
  728.         for ($i = $max_length; $i < $note_length; $i += $max_length) {
  729.             $chunk  = substr($note, $i, $max_length);
  730.             $length = 0x0006 + strlen($chunk);
  731.             $header = pack("vv",   $record, $length);
  732.             $data   = pack("vvv", -1, 0, strlen($chunk));
  733.             $this->_append($header.$data.$chunk);
  734.         }
  735.         return(0);
  736.     }
  737.  
  738.     /**
  739.      * Write a blank cell to the specified row and column (zero indexed).
  740.      * A blank cell is used to specify formatting without adding a string
  741.      * or a number.
  742.      *
  743.      * A blank cell without a format serves no purpose. Therefore, we don't write
  744.      * a BLANK record unless a format is specified.
  745.      *
  746.      * Returns  0 : normal termination (including no format)
  747.      *         -1 : insufficient number of arguments
  748.      *         -2 : row or column out of range
  749.      *
  750.      * @param integer $row    Zero indexed row
  751.      * @param integer $col    Zero indexed column
  752.      * @param mixed   $format The XF format
  753.      */
  754.     function _writeBlank($row, $col, $xfIndex)
  755.     {
  756.         $record    = 0x0201;                 // Record identifier
  757.         $length    = 0x0006;                 // Number of bytes to follow
  758.  
  759.         $header    = pack("vv",  $record, $length);
  760.         $data      = pack("vvv", $row, $col, $xfIndex);
  761.         $this->_append($header . $data);
  762.         return 0;
  763.     }
  764.  
  765.     /**
  766.      * Write a boolean or an error type to the specified row and column (zero indexed)
  767.      *
  768.      * @param int $row Row index (0-based)
  769.      * @param int $col Column index (0-based)
  770.      * @param int $value 
  771.      * @param boolean $isError Error or Boolean?
  772.      * @param int $xfIndex 
  773.      */
  774.     private function _writeBoolErr($row, $col, $value, $isError, $xfIndex)
  775.     {
  776.         $record = 0x0205;
  777.         $length = 8;
  778.  
  779.         $header    = pack("vv",  $record, $length);
  780.         $data      = pack("vvvCC", $row, $col, $xfIndex, $value, $isError);
  781.         $this->_append($header . $data);
  782.         return 0;
  783.     }
  784.  
  785.     /**
  786.      * Write a formula to the specified row and column (zero indexed).
  787.      * The textual representation of the formula is passed to the parser in
  788.      * Parser.php which returns a packed binary string.
  789.      *
  790.      * Returns  0 : normal termination
  791.      *         -1 : formula errors (bad formula)
  792.      *         -2 : row or column out of range
  793.      *
  794.      * @param integer $row     Zero indexed row
  795.      * @param integer $col     Zero indexed column
  796.      * @param string  $formula The formula text string
  797.      * @param mixed   $format  The optional XF format
  798.      * @param mixed   $calculatedValue  Calculated value
  799.      * @return integer 
  800.      */
  801.     private function _writeFormula($row, $col, $formula, $xfIndex, $calculatedValue)
  802.     {
  803.         $record    = 0x0006;     // Record identifier
  804.  
  805.         // Initialize possible additional value for STRING record that should be written after the FORMULA record?
  806.         $stringValue = null;
  807.  
  808.         // calculated value
  809.         if (isset($calculatedValue)) {
  810.  
  811.             // Since we can't yet get the data type of the calculated value,
  812.             // we use best effort to determine data type
  813.  
  814.             if (is_bool($calculatedValue)) {
  815.                 // Boolean value
  816.                 $num = pack('CCCvCv', 0x01, 0x00, (int)$calculatedValue, 0x00, 0x00, 0xFFFF);
  817.  
  818.             } elseif (is_int($calculatedValue) || is_float($calculatedValue)) {
  819.                 // Numeric value
  820.                 $num = pack('d', $calculatedValue);
  821.  
  822.             } elseif (is_string($calculatedValue)) {
  823.                 if (array_key_exists($calculatedValue, PHPExcel_Cell_DataType::getErrorCodes())) {
  824.                     // Error value
  825.                     $num = pack('CCCvCv', 0x02, 0x00, $this->_mapErrorCode($calculatedValue), 0x00, 0x00, 0xFFFF);
  826.  
  827.                 } elseif ($calculatedValue === '' && $this->_BIFF_version == 0x0600) {
  828.                     // Empty string (and BIFF8)
  829.                     $num = pack('CCCvCv', 0x03, 0x00, 0x00, 0x00, 0x00, 0xFFFF);
  830.  
  831.                 } else {
  832.                     // Non-empty string value (or empty string BIFF5)
  833.                     $stringValue = $calculatedValue;
  834.                     $num = pack('CCCvCv', 0x00, 0x00, 0x00, 0x00, 0x00, 0xFFFF);
  835.  
  836.                 }
  837.  
  838.             } else {
  839.                 // We are really not supposed to reach here
  840.                 $num = pack('d', 0x00);
  841.  
  842.             }
  843.  
  844.         } else {
  845.             $num = pack('d', 0x00);
  846.         }
  847.  
  848.         $grbit     = 0x03;                // Option flags
  849.         $unknown   = 0x0000;              // Must be zero
  850.  
  851.         // Strip the '=' or '@' sign at the beginning of the formula string

  854.         } else {

  859.

  867.

  869.  
  870.             $data      = pack("vvv", $row, $col, $xfIndex)
  871.                         . $num
  872.                         . pack("vVv", $grbit, $unknown, $formlen);

  877.                 $this->_writeStringRecord($stringValue);
  878.             }
  879.  
  880.             return 0;
  881.  
  882.         } catch (Exception $e) {
  883.             // do nothing
  884.         }
  885.  
  886.     }

  1068.      * @param integer $col2   End column

  1070.      * @return integer 
  1071.      */

  1073.     {

  1075.         // MS/Novell network drives and shares start with \\

  1092.

  1094.         if ( preg_match('/^[A-Z]:/', $url) ) {
  1095.             $absolute = 0x02; // absolute path on Windows, e.g. C:\...
  1096.         }

  1104.             $link_type |= 0x08;
  1105.         }
  1106.  
  1107.  
  1108.         // Pack the link type

  1117.

  1119.         $dir_long       = $dir_long . "\0";
  1120.  
  1121.         // Pack the lengths of the dir strings
  1122.         $dir_short_len = pack("V", strlen($dir_short)      );
  1123.         $dir_long_len  = pack("V", strlen($dir_long)       );
  1124.         $stream_len    = pack("V", 0);//strlen($dir_long) + 0x06);
  1125.  
  1126.         // Pack the undocumented parts of the hyperlink stream

  1129.         $unknown3 = pack("H*",'FFFFADDE000000000000000000000000000000000000000');

  1131.  
  1132.         // Pack the main data stream
  1133.         $data        = pack("vvvv", $row1, $row2, $col1, $col2) .
  1134.                           $unknown1     .
  1135.                           $link_type    .
  1136.                           $unknown2     .
  1137.                           $up_count     .
  1138.                           $dir_short_len.
  1139.                           $dir_short    .
  1140.                           $unknown3     .
  1141.                           $stream_len   ;/*.
  1142.                           $dir_long_len .
  1143.                           $unknown4     .
  1144.                           $dir_long     .
  1145.                           $sheet_len    .
  1146.                           $sheet        ;*/
  1147.  
  1148.         // Pack the header data
  1149.         $length   = strlen($data);
  1150.         $header   = pack("vv", $record, $length);
  1151.  
  1152.         // Write the packed data
  1153.         $this->_append($header. $data);
  1154.         return 0;
  1155.     }

  1157.     /**
  1158.      * This method is used to set the height and format for a row.
  1159.      *
  1160.      * @param integer $row    The row to set

  1163.      * @param integer $xfIndex  The optional cell style Xf index to apply to the columns
  1164.      * @param bool    $hidden The optional hidden attribute
  1165.      * @param integer $level  The optional outline level for row, in range [0,7]

  1167.     private function _writeRow($row, $height, $xfIndex, $hidden = false, $level = 0)
  1168.     {

  1170.         $length      = 0x0010;               // Number of bytes to follow
  1171.

  1173.         $colMac      = 0x0000;               // Last defined column
  1174.         $irwMac      = 0x0000;               // Used by Excel to optimise loading
  1175.         $reserved    = 0x0000;               // Reserved
  1176.         $grbit       = 0x0000;               // Option flags
  1177.         $ixfe        = $xfIndex;
  1178.

  1180.             $height = null;
  1181.         }

  1184.         if ($height != null) {
  1185.             $miyRw = $height * 20;  // row height
  1186.         } else {

  1188.         }
  1189.  
  1190.         // Set the options flags. fUnsynced is used to show that the font and row

  1192.         // The collapsed flag 0x10 doesn't seem to be used to indicate that a row

  1199.         }

  1201.             $grbit |= 0x0040; // fUnsynced
  1202.         }

  1204.             $grbit |= 0x0080;
  1205.         }
  1206.         $grbit |= 0x0100;

  1210.                                      $irwMac,$reserved, $grbit, $ixfe);

  1212.     }

  1214.     /**
  1215.      * Writes Excel DIMENSIONS to define the area in which there is data.
  1216.      */
  1217.     private function _writeDimensions()
  1218.     {
  1219.         $record = 0x0200; // Record identifier
  1220.  
  1221.         if ($this->_BIFF_version == 0x0500) {
  1222.             $length = 0x000A;               // Number of bytes to follow
  1223.             $data = pack("vvvvv"
  1224.                     , $this->_firstRowIndex
  1225.                     , $this->_lastRowIndex + 1

  1731.             if (!isset($rwTop)) {

  1755.

  2403.      * function. The values of $col_end and $row_end are calculated by subtracting
  2404.      * the width and height of the bitmap from the width and height of the
  2405.      * underlying cells.
  2406.      * The vertices are expressed as a percentage of the underlying cell width as
  2407.      * follows (rhs values are in pixels):
  2408.      *
  2409.      *       x1 = X / W *1024
  2410.      *       y1 = Y / H *256

  2412.      *       y2 = (Y-1) / H *256
  2413.      *
  2414.      *       Where:  X is distance from the left side of the underlying cell
  2415.      *               Y is distance from the top of the underlying cell
  2416.      *               W is the width of the cell
  2417.      *               H is the height of the cell

  2419.      *        percentage of 1024.
  2420.      *
  2421.      * @access private
  2422.      * @param integer $col_start Col containing upper left corner of object
  2423.      * @param integer $row_start Row containing top left corner of object
  2424.      * @param integer $x1        Distance to left side of object
  2425.      * @param integer $y1        Distance to top of object
  2426.      * @param integer $width     Width of image frame
  2427.      * @param integer $height    Height of image frame

  2429.     function _positionImage($col_start, $row_start, $x1, $y1, $width, $height)

  2459.         // with zero eight or width.
  2460.         //

  2462.             return;

  2484.     }

  2593.     /**
  2594.      * Convert a 24 bit bitmap into the modified internal format used by Windows.
  2595.      * This is described in BITMAPCOREHEADER and BITMAPCOREINFO structures in the
  2596.      * MSDN library.
  2597.      *
  2598.      * @access private
  2599.      * @param string $bitmap The bitmap to process
  2600.      * @return array Array with data and properties of the bitmap
  2601.      */
  2602.     function _processBitmap($bitmap)
  2603.     {
  2604.         // Open file.
  2605.         $bmp_fd = @fopen($bitmap,"rb");
  2606.         if (!$bmp_fd) {
  2607.             throw new Exception("Couldn't import $bitmap");
  2608.         }
  2609.  
  2610.         // Slurp the file into a string.
  2611.         $data = fread($bmp_fd, filesize($bitmap));
  2612.  
  2613.         // Check that the file is big enough to be a bitmap.
  2614.         if (strlen($data) <= 0x36) {
  2615.             throw new Exception("$bitmap doesn't contain enough data.\n");
  2616.         }
  2617.  
  2618.         // The first 2 bytes are used to identify the bitmap.
  2619.         $identity = unpack("A2ident", $data);
  2620.         if ($identity['ident'] != "BM") {
  2621.             throw new Exception("$bitmap doesn't appear to be a valid bitmap image.\n");
  2622.         }
  2623.  
  2624.         // Remove bitmap data: ID.
  2625.         $data = substr($data, 2);
  2626.  
  2627.         // Read and remove the bitmap size. This is more reliable than reading
  2628.         // the data size at offset 0x22.
  2629.         //
  2630.         $size_array   = unpack("Vsa", substr($data, 0, 4));
  2631.         $size   = $size_array['sa'];
  2632.         $data   = substr($data, 4);
  2633.         $size  -= 0x36; // Subtract size of bitmap header.
  2634.         $size  += 0x0C; // Add size of BIFF header.
  2635.  
  2636.         // Remove bitmap data: reserved, offset, header length.
  2637.         $data = substr($data, 12);
  2638.  
  2639.         // Read and remove the bitmap width and height. Verify the sizes.
  2640.         $width_and_height = unpack("V2", substr($data, 0, 8));
  2641.         $width  = $width_and_height[1];
  2642.         $height = $width_and_height[2];
  2643.         $data   = substr($data, 8);
  2644.         if ($width > 0xFFFF) {
  2645.             throw new Exception("$bitmap: largest image width supported is 65k.\n");
  2646.         }
  2647.         if ($height > 0xFFFF) {
  2648.             throw new Exception("$bitmap: largest image height supported is 65k.\n");
  2649.         }
  2650.  
  2651.         // Read and remove the bitmap planes and bpp data. Verify them.
  2652.         $planes_and_bitcount = unpack("v2", substr($data, 0, 4));
  2653.         $data = substr($data, 4);
  2654.         if ($planes_and_bitcount[2] != 24) { // Bitcount
  2655.             throw new Exception("$bitmap isn't a 24bit true color bitmap.\n");
  2656.         }
  2657.         if ($planes_and_bitcount[1] != 1) {
  2658.             throw new Exception("$bitmap: only 1 plane supported in bitmap image.\n");
  2659.         }
  2660.  
  2661.         // Read and remove the bitmap compression. Verify compression.
  2662.         $compression = unpack("Vcomp", substr($data, 0, 4));
  2663.         $data = substr($data, 4);
  2664.  
  2665.         //$compression = 0;
  2666.         if ($compression['comp'] != 0) {
  2667.             throw new Exception("$bitmap: compression not supported in bitmap image.\n");
  2668.         }
  2669.  
  2670.         // Remove bitmap data: data size, hres, vres, colours, imp. colours.
  2671.         $data = substr($data, 20);
  2672.  
  2673.         // Add the BITMAPCOREHEADER data
  2674.         $header  = pack("Vvvvv", 0x000c, $width, $height, 0x01, 0x18);
  2675.         $data    = $header . $data;
  2676.  
  2677.         return (array($width, $height, $size, $data));
  2678.     }
