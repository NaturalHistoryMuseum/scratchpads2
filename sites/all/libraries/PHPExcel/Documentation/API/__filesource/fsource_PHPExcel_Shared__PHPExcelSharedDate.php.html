  80.     public static function ExcelToPHP($dateValue = 0) {
  81.         if (self::$ExcelBaseDate == self::CALENDAR_WINDOWS_1900) {
  82.             $myExcelBaseDate = 25569;
  83.             //    Adjust for the spurious 29-Feb-1900 (Day 60)
  84.             if ($dateValue < 60) {
  85.                 --$myExcelBaseDate;
  86.             }
  87.         } else {
  88.             $myExcelBaseDate = 24107;
  89.         }
  90.  
  91.         // Perform conversion
  92.         if ($dateValue >= 1) {
  93.             $utcDays = $dateValue - $myExcelBaseDate;
  94.             $returnValue = round($utcDays * 24 * 60 * 60);
  95.             if (($returnValue <= PHP_INT_MAX) && ($returnValue >= -PHP_INT_MAX)) {
  96.                 $returnValue = (integer) $returnValue;
  97.             }
  98.         } else {
  99.             $hours = round($dateValue * 24);
  100.             $mins = round($dateValue * 24 * 60) - round($hours * 60);
  101.             $secs = round($dateValue * 24 * 60 * 60) - round($hours * 60 * 60) - round($mins * 60);
  102.             $returnValue = (integer) gmmktime($hours, $mins, $secs);
  103.         }
  104.  
  105.         // Return
  106.         return $returnValue;
  107.     }    //    function ExcelToPHP()

  117.         $dateTime = self::ExcelToPHP($dateValue);
  118.         $days = floor($dateTime / 86400);

  124.         $dateObj = date_create('1-Jan-1970+'.$days.' days');
  125.         $dateObj->setTime($hours,$minutes,$seconds);

  127.         return $dateObj;

  157.     /**
  158.      * FormattedPHPToExcel
  159.      *
  160.      * @param    long    $year 
  161.      * @param    long    $month 
  162.      * @param    long    $day 
  163.      * @param    long    $hours 
  164.      * @param    long    $minutes 
  165.      * @param    long    $seconds 
  166.      * @return  long                Excel date/time value
  167.      */
  168.     public static function FormattedPHPToExcel($year, $month, $day, $hours=0, $minutes=0, $seconds=0) {
  169.         if (self::$ExcelBaseDate == self::CALENDAR_WINDOWS_1900) {
  170.             //
  171.             //    Fudge factor for the erroneous fact that the year 1900 is treated as a Leap Year in MS Excel
  172.             //    This affects every date following 28th February 1900
  173.             //
  174.             $excel1900isLeapYear = True;
  175.             if (($year == 1900) && ($month <= 2)) { $excel1900isLeapYear = False; }
  176.             $myExcelBaseDate = 2415020;
  177.         } else {
  178.             $myExcelBaseDate = 2416481;
  179.             $excel1900isLeapYear = False;
  180.         }
  181.  
  182.         //    Julian base date Adjustment
  183.         if ($month > 2) {
  184.             $month = $month - 3;
  185.         } else {
  186.             $month = $month + 9;
  187.             --$year;
  188.         }
  189.  
  190.         //    Calculate the Julian Date, then subtract the Excel base date (JD 2415020 = 31-Dec-1899 Giving Excel Date of 0)
  191.         $century = substr($year,0,2);
  192.         $decade = substr($year,2,2);
  193.         $excelDate = floor((146097 * $century) / 4) + floor((1461 * $decade) / 4) + floor((153 * $month + 2) / 5) + $day + 1721119 - $myExcelBaseDate + $excel1900isLeapYear;
  194.  
  195.         $excelTime = (($hours * 3600) + ($minutes * 60) + $seconds) / 86400;
  196.  
  197.         return (float) $excelDate + $excelTime;
  198.     }    //    function FormattedPHPToExcel()

  231.     public static function isDateTimeFormatCode($pFormatCode = '') {
  232.         // Switch on formatcode
  233.         switch ($pFormatCode) {
  234.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_YYYYMMDD:
  235.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_YYYYMMDD2:
  236.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_DDMMYYYY:
  237.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_DMYSLASH:
  238.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_DMYMINUS:
  239.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_DMMINUS:
  240.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_MYMINUS:
  241.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_DATETIME:
  242.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_TIME1:
  243.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_TIME2:
  244.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_TIME3:
  245.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_TIME4:
  246.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_TIME5:
  247.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_TIME6:
  248.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_TIME7:
  249.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_TIME8:
  250.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_YYYYMMDDSLASH:
  251.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_XLSX14:
  252.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_XLSX15:
  253.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_XLSX16:
  254.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_XLSX17:
  255.             case PHPExcel_Style_NumberFormat::FORMAT_DATE_XLSX22:
  256.                 return true;
  257.         }
  258.  
  259.         //    Typically number, currency or accounting (or occasionally fraction) formats
  260.         if ((substr($pFormatCode,0,1) == '_') || (substr($pFormatCode,0,2) == '0 ')) {
  261.             return false;
  262.         }
  263.         // Try checking for any of the date formatting characters that don't appear within square braces
  264.         if (preg_match('/(^|\])[^\[]*['.self::$possibleDateFormatCharacters.']/i',$pFormatCode)) {
  265.             //    We might also have a format mask containing quoted strings...
  266.             //        we don't want to test for any of our characters within the quoted blocks
  267.             if (strpos($pFormatCode,'"') !== false) {
  268.                 $i = false;
  269.                 foreach(explode('"',$pFormatCode) as $subVal) {
  270.                     //    Only test in alternate array entries (the non-quoted blocks)
  271.                     if (($i = !$i) && (preg_match('/(^|\])[^\[]*['.self::$possibleDateFormatCharacters.']/i',$subVal))) {
  272.                         return true;
  273.                     }
  274.                 }
  275.                 return false;
  276.             }
  277.             return true;
  278.         }
  279.  
  280.         // No date...
  281.         return false;
  282.     }    //    function isDateTimeFormatCode()

  286.      * Convert a date/time string to Excel time
  287.      *
  288.      * @param    string    $dateValue        Examples: '2009-12-31', '2009-12-31 15:59', '2009-12-31 15:59:10'
  289.      * @return    float|false       Excel date/time serial value
  290.      */
  291.     public static function stringToExcel($dateValue = '') {
  292.         if (strlen($dateValue) < 2)
  293.             return false;
  294.         if (!preg_match('/^(\d{1,4}[ \.\/\-][A-Z]{3,9}([ \.\/\-]\d{1,4})?|[A-Z]{3,9}[ \.\/\-]\d{1,4}([ \.\/\-]\d{1,4})?|\d{1,4}[ \.\/\-]\d{1,4}([ \.\/\-]\d{1,4})?)( \d{1,2}:\d{1,2}(:\d{1,2})?)?$/iu', $dateValue))
  295.             return false;
  296.  
  297.         $dateValueNew = PHPExcel_Calculation_DateTime::DATEVALUE($dateValue);
  298.  
  299.         if ($dateValueNew === PHPExcel_Calculation_Functions::VALUE()) {
  300.             return false;
  301.         } else {
  302.             if (strpos($dateValue, ':') !== false) {
  303.                 $timeValue = PHPExcel_Calculation_DateTime::TIMEVALUE($dateValue);
  304.                 if ($timeValue === PHPExcel_Calculation_Functions::VALUE()) {
  305.                     return false;
  306.                 }
  307.                 $dateValueNew += $timeValue;
  308.             }
  309.             return $dateValueNew;
  310.         }
